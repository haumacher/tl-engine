Ausprogrammierte ExcelActions

ExcelActionOp können implementiert werden und in der ExcelActionRegistry registriert werden:

<config config:interface="com.top_logic.layout.scripting.template.excel.ExcelActionRegistry$Config">
	<registrations>
		<object-type name="Projekt">
			<actions>
				<action name="Umbenennen"
					implementation="com.top_logic.demo.layout.scripting.action.excel.RenameElementOp" />
				<action name="Löschen"
					implementation="com.top_logic.demo.layout.scripting.action.excel.DeleteElementOp" />
			</actions>
		</object-type>
		<object-type name="Kontakt">
			<actions>
				<action name="Anlegen"
					implementation="com.top_logic.demo.layout.scripting.action.excel.DeleteElementOp" />
			</actions>
		</object-type>
	</registrations>
</config>

Diese haben Priorität vor den XML-Templates, damit es möglich ist zB. aus der Basis geerbte XML-Templates zu durch ExcelActionOp zu überschreiben. Soll hingegen eine geerbte ExcelActionOp durch ein XML-Template überschrieben werden, muss die ExcelActionOp mittels config:override="true" bzw. config:operation="remove" deregistriert werden:

<config config:interface="com.top_logic.layout.scripting.template.excel.ExcelActionRegistry$Config">
	<registrations>
		<object-type name="Projekt">
			<actions>
				<action name="Löschen" config:operation="remove" />
			</actions>
		</object-type>
		<object-type name="Kontakt" config:override="true" />
	</registrations>
</config>
